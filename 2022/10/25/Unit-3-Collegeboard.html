public boolean isBalanced(ArrayList<String> delimiters) {
    int open = 0;
    int close = 0;
    for (String d: delimiters) {
        if (d.equals(openDel)) {
            open++;
        } else if (d.equals(closeDel)) {
            close++;
        }

        if (close > open) return false; 
    }
    return open == close; 
}
