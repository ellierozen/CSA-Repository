  • Uses the operands AND, OR and NOT and two input variables, A and B

boolean b = false;
if (a) {
    System.out.println("True code block");
}

if (a && !b) {
    System.out.println("True code block");

if (a || b) {
    System.out.println("True or False code block");
}

if ((a && !b) && (a || a)) {
